n = int(input())
matrix1 = [[int(i) for i in input().split()] for _ in range(n)]
matrix2 = matrix1
k = int(input())-1
while k!=0:
    matrix = [[] for _ in range(n)]
    for z in range(n):
        for i in range(n):
            element = 0
            for j in range(n):
                element = element + matrix1[z][j]*matrix2[j][i]
            matrix[z].append(element)
    matrix1 = matrix
    k-=1
for i in range(n):
    for j in range(n):
        print(matrix[i][j],end = ' ')
    print()
